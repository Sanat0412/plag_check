import java.util.Scanner;

public class Main {
	static Scanner s = new Scanner(System.in);
	public static void main(String[] args){
		int T = s.nextInt();
		while(T-- != 0)
		    solve();
		s.close();
	}
	public static void solve(){
		int n = s.nextInt(), k = s.nextInt(), l = 0, r = 0;
		for(int i = 0; i < n; ++i){
			int a = s.nextInt(), b = s.nextInt();
			if(a == k)
				l++;
			if(b == k)
				r++;
		}
		System.out.println(l != 0 && r != 0? "YES" : "NO");
	}
}
